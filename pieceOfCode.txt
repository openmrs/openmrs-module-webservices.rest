	private Path mergePaths(Path path1, Path path2) {
		
		Path mergedPath = new Path();
		Map<String, Operation> path1Operation = path1.getOperations();
		Map<String, Operation> path2Operation = path2.getOperations();
		
		Map<String, Operation> mergedOperations = new HashMap<String, Operation>();
		
		for (String key : path1Operation.keySet())
			mergedOperations.put(key, path1Operation.get(key));
		
		for (String key : path2Operation.keySet())
			mergedOperations.put(key, path2Operation.get(key));
		
		mergedPath.setOperations(mergedOperations);
		
		return mergedPath;
		
	}
